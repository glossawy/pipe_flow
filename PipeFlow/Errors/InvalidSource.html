def self.msg_for(node)
  case node
  when AST::MethodCall
    'An incomplete method call is a valid right-hand side of a pipeline but not ' \
    "a valid left-hand side (#{node} >> ... is invalid)"
  else
    "#{node.class.name} is unexpected on the left-hand side of a pipeline " \
    "(#{node} >> ... is unexpected)"
  end
end
